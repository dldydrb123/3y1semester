# 초기 위치, 속도, 가속도 설정
initial_position = vector(-5, 0, 0)
initial_velocity = vector(0, 0, 0)
acceleration = vector(1, 0, 0)

# 시간 간격 설정
dt = 0.01

# 상자 생성
box(size=vector(0.5, 0.25, 0.25), color=color.yellow, pos=initial_position)

# 상자의 속도와 위치 초기화
box_velocity = initial_velocity
box_position = initial_position

# 시간 초기화
t = 0

# 종료 시간 설정
end_time = 5

# 등가속도 운동 루프
while t <= end_time:
    rate(100)  # 프레임 속도 조절
    
    # 등가속도 운동 방정식 적용
    box_velocity += acceleration * dt
    box_position += box_velocity * dt
    
    # 상자의 위치 업데이트
    box.pos = box_position
    
    # 시간 업데이트
    t += dt
